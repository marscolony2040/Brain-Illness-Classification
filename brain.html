Brain Illness Scanner with PyTorch¶

By. Mohammed Sharieff¶

Description¶

I used Google Gemini to generate a high-fidelity synthetic dataset covering four distinct brain states: Normal, Trauma, Tumor, and Schizophrenia. I then engineered a custom CNN in PyTorch to handle feature extraction and classify these neurological patterns. To keep the project modular and reproducible, I decoupled the logic into a standalone trainer.py for model training. Finally, I loaded the model into a notebook to run batch inference and visualize how the predictions mapped against the ground-truth scans.

transform = transforms.Compose([

    transforms.Resize((300, 300)),
    transforms.ToTensor()
])

# Loads the outputs from the training set

# Loads the test case images to be classified and transforms images to torch tensor
tests = [Image.open(f'testcases/{i}.png') for i in range(8)]
itests = [transform(u) for u in tests]

# Loads the actual outputs from the images being predicted

# Contains brain illness choices to be referenced and indexed during testing

'''
Convolutional Neural Network for Image Classification

- Kernel Size: Dimensions of the sliding feature-extraction window.

- Stride: Pixel distance the window moves per step.

- Padding: Extra border pixels to preserve edge data.
'''

class Predict(nn.Module):
    def __init__(self):
        super(Predict, self).__init__()

        # Sets CNN Layers and Activation Functions (ReLU)
        self.moving = nn.Sequential(
            nn.Conv2d(3, 64, kernel_size=3, stride=1, padding=1),
            nn.ReLU(),
            nn.Conv2d(64, 128, kernel_size=3, stride=2, padding=1),
            nn.ReLU(),
            nn.Conv2d(128, 300, kernel_size=3, stride=2, padding=1),
            nn.ReLU(),
            nn.Conv2d(300, 150, kernel_size=3, stride=1, padding=1),
            nn.AdaptiveAvgPool2d((1, 1))
        )
        
        # Classifies all 52 images
        self.ending = nn.Sequential(
            nn.Linear(150, len(outputs))
        )
    
    # Forward propigation function
    def forward(self, x):
        # Run inputs through CNN
        x = self.moving(x)

        # Flatten inputs for Classifier
        x = x.flatten(1)

        # Return Classifications
        return self.ending(x)

# Loads pre-trained model
model = torch.load('brain_model.pth')

# Renders Matplotlib Figure for plotting images
fig = plt.figure()

# Declares subplot for splitting images
ax = [fig.add_subplot(2, 4, i+1) for i in range(8)]

# Runs eight test cases and plots their predictions with the actual classifications
for i in range(8):

    # Ensures model gradients are not altered with each prediction
    with torch.no_grad():
        testout = model(itests[i].unsqueeze(0))

    # Extract the index of the output list which contains the classification
    s1 = torch.softmax(testout, dim=1)
    p1 = torch.argmax(s1, dim=1).item()
    
    # Load predicted and actual for comparison
    predicted_choice = choices[outputs[p1]]
    actual_choice = choices[ioutputs[i]]

    # Create a plot title which shows the predicted and actual classifications
    title = f'Predicted Issue: {predicted_choice}\nActual Issue: {actual_choice}'

    # Plot the image
    ax[i].cla()
    ax[i].set_title(title, fontsize=7)
    ax[i].imshow(tests[i].convert("RGB"))
    ax[i].axis('off')

plt.show()
